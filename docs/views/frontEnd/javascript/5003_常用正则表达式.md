```javascript
// 至少含有两个不连续的 :
const reg = /^[^:]{2,}(?::[^:]+){2,}$/gm;

// 提取 xxx-xxx-xxx 格式的最后一组字符
function extractText(text) {
    let str = '';
    if (/([^-]+)$/.test(text)) {
        str = RegExp.$1;
    }
    return str || text;
}
```
